\subsection{Step Resolution Using Measured Values}

\subsubsection{Resolution at 12VDC}

\begin{itemize}
    \item [] Measured Frequency = 641.0hz or cycles per second
    \item [] Measured RPM = 5,500RPM or revolutions per minute
    \begin{itemize}
        \item [] Convert RPM to revolutions per second
        \begin{itemize}
            \item [] $RPS=\frac{RPM}{60}=\frac{5,500}{60}=91.667RPS$
        \end{itemize}
        \item[] Set cycles per second equal to revolutions per second and solve for the resolution per clock cycle. 
        \begin{itemize}
            \item [] $\frac{641cycles}{Second}= \frac{91.667revolutions}{Second}$
            \item[] $641cycles=91.667revolutions$
            \item[] $1cycle=\frac{91.667revolutions}{641}=0.143revolutions$
        \end{itemize}
        \item[] Convert revolutions to degrees
        \begin{itemize}
            \item [] $1rev=360^{\circ}$
            \item[] $1cycle=\frac{91.667revolutions}{641}=0.143\times 360^{\circ}=51.48^{\circ}$
        \end{itemize}
    \end{itemize}
    \item[] Final Formula for converting measured RPM and Frequency to Step Resolution:
\end{itemize}

    \begin{equation}
    Step \ Resolution  = \left(\frac{RPM}{Frequency}\times 6 \right)^{\circ} 
    \end{equation}
